276// Macro to enable floating point precise mode and to disable fused multiply add instructions
277#if defined(JPH_COMPILER_CLANG) || defined(JPH_COMPILER_GCC)
278 // In clang it appears you cannot turn off -ffast-math and -ffp-contract=fast for a code block
279 // There is #pragma clang fp contract (off) but that doesn't seem to work under clang 9 & 10 when -ffast-math is specified on the commandline (you override it to turn it on, but not off)
280 // There is #pragma float_control(precise, on) but that doesn't work under clang 9.
281 // So for now we compile clang without -ffast-math so the macros are empty
282 #define JPH_PRECISE_MATH_ON
283 #define JPH_PRECISE_MATH_OFF
284#elif defined(JPH_COMPILER_MSVC)
285 // Unfortunately there is no way to push the state of fp_contract, so we have to assume it was turned on before JPH_PRECISE_MATH_ON
286 #define JPH_PRECISE_MATH_ON \
287 __pragma(float_control(precise, on, push)) \
288 __pragma(fp_contract(off))
289 #define JPH_PRECISE_MATH_OFF \
290 __pragma(fp_contract(on)) \
291 __pragma(float_control(pop))
292#else
293 #error Undefined
294#endif
