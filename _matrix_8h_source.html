234template <>
235inline bool Matrix<2, 2>::SetInversed(const Matrix<2, 2> &inM)
236{
237 // Fetch elements
238 float a = inM.mCol[0].mF32[0];
239 float b = inM.mCol[1].mF32[0];
240 float c = inM.mCol[0].mF32[1];
241 float d = inM.mCol[1].mF32[1];
242
243 // Calculate determinant
244 float det = a * d - b * c;
245 if (det == 0.0f)
246 return false;
247
248 // Construct inverse
249 mCol[0].mF32[0] = d / det;
250 mCol[1].mF32[0] = -b / det;
251 mCol[0].mF32[1] = -c / det;
252 mCol[1].mF32[1] = a / det;
253 return true;
254}
