194 inline bool SetBarrier(Barrier *inBarrier)
195 {
196 intptr_t barrier = 0;
197 if (mBarrier.compare_exchange_strong(barrier, reinterpret_cast<intptr_t>(inBarrier), memory_order_relaxed))
198 return true;
199 JPH_ASSERT(barrier == cBarrierDoneState, "A job can only belong to 1 barrier");
200 return false;
201 }
202
204 inline uint32 Execute()
205 {
206 // Transition job to executing state
207 uint32 state = 0; // We can only start running with a dependency counter of 0
208 if (!mNumDependencies.compare_exchange_strong(state, cExecutingState, memory_order_acquire))
209 return state; // state is updated by compare_exchange_strong to the current value
210
211 // Run the job function
212 {
213 JPH_PROFILE(mJobName, mColor.GetUInt32());
214 mJobFunction();
215 }
216
217 // Fetch the barrier pointer and exchange it for the done state, so we're sure that no barrier gets set after we want to call the callback
218 intptr_t barrier = mBarrier.load(memory_order_relaxed);
219 for (;;)
220 {
221 if (mBarrier.compare_exchange_weak(barrier, cBarrierDoneState, memory_order_relaxed))
222 break;
223 }
224 JPH_ASSERT(barrier != cBarrierDoneState);
225
226 // Mark job as done
227 state = cExecutingState;
228 mNumDependencies.compare_exchange_strong(state, cDoneState, memory_order_relaxed);
229 JPH_ASSERT(state == cExecutingState);
230
231 // Notify the barrier after we've changed the job to the done state so that any thread reading the state after receiving the callback will see that the job has finished
232 if (barrier != 0)
233 reinterpret_cast<Barrier *>(barrier)->OnJobFinished(this);
234
235 return cDoneState;
236 }
237
239 inline bool CanBeExecuted() const { return mNumDependencies.load(memory_order_relaxed) == 0; }

242 inline bool IsDone() const { return mNumDependencies.load(memory_order_relaxed) == cDoneState; }
