14inline uint64 HashBytes(const void *inData, uint inSize, uint64 inSeed = 0xcbf29ce484222325UL)
15{
16 uint64 hash = inSeed;
17 for (const uint8 *data = reinterpret_cast<const uint8 *>(inData); data < reinterpret_cast<const uint8 *>(inData) + inSize; ++data)
18 {
19 hash = hash ^ uint64(*data);
20 hash = hash * 0x100000001b3UL;
21 }
22 return hash;
23}
