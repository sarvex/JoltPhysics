30class TempAllocatorImpl final : public TempAllocator
31{
32public:
33 JPH_OVERRIDE_NEW_DELETE
34
36 explicit TempAllocatorImpl(uint inSize) :
37 mBase(static_cast<uint8 *>(JPH::Allocate(inSize))),
38 mSize(inSize)
39 {
40 }
41
43 virtual ~TempAllocatorImpl() override
44 {
45 JPH_ASSERT(mTop == 0);
46 JPH::Free(mBase);
47 }
48
49 // See: TempAllocator
50 virtual void * Allocate(uint inSize) override
51 {
52 if (inSize == 0)
53 {
54 return nullptr;
55 }
56 else
57 {
58 uint new_top = mTop + AlignUp(inSize, 16);
59 if (new_top > mSize)
60 JPH_CRASH; // Out of memory
61 void *address = mBase + mTop;
62 mTop = new_top;
63 return address;
64 }
65 }
66
67 // See: TempAllocator
68 virtual void Free(void *inAddress, uint inSize) override
69 {
70 if (inAddress == nullptr)
71 {
72 JPH_ASSERT(inSize == 0);
73 }
74 else
75 {
76 mTop -= AlignUp(inSize, 16);
77 if (mBase + mTop != inAddress)
78 JPH_CRASH; // Freeing in the wrong order
79 }
80 }
81
82 // Check if no allocations have been made
83 bool IsEmpty() const
84 {
85 return mTop == 0;
86 }
87
88private:
89 uint8 * mBase;
90 uint mSize;
91 uint mTop = 0;
92};
