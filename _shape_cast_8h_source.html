105 // Calculate by how much to shift the contact points

107
108 ShapeCastResult result;
109 result.mContactPointOn2 = mContactPointOn1 - delta;
110 result.mContactPointOn1 = mContactPointOn2 - delta;

112 result.mPenetrationDepth = mPenetrationDepth;
113 result.mSubShapeID2 = mSubShapeID1;
114 result.mSubShapeID1 = mSubShapeID2;
115 result.mBodyID2 = mBodyID2;
116 result.mFraction = mFraction;
117 result.mIsBackFaceHit = mIsBackFaceHit;
118
119 result.mShape2Face.resize(mShape1Face.size());
120 for (Face::size_type i = 0; i < mShape1Face.size(); ++i)
121 result.mShape2Face[i] = mShape1Face[i] - delta;
122
123 result.mShape1Face.resize(mShape2Face.size());
124 for (Face::size_type i = 0; i < mShape2Face.size(); ++i)
125 result.mShape1Face[i] = mShape2Face[i] - delta;
