170template <class T>
171bool OSReadData(ObjectStreamIn &ioStream, vector<T> &inArray)
172{
173 bool continue_reading = true;
174
175 // Read array length
176 uint32 array_length;
177 continue_reading = ioStream.ReadCount(array_length);
178
179 // Read array items
180 if (continue_reading)
181 {
182 inArray.resize(array_length);
183 for (uint32 el = 0; el < array_length && continue_reading; ++el)
184 continue_reading = OSReadData(ioStream, inArray[el]);
185 }
186
187 return continue_reading;
188}

191template <class T, uint N>
192bool OSReadData(ObjectStreamIn &ioStream, StaticArray<T, N> &inArray)
193{
194 bool continue_reading = true;
195
196 // Read array length
197 uint32 array_length;
198 continue_reading = ioStream.ReadCount(array_length);
199
200 // Check if we can fit this many elements
201 if (array_length > N)
202 return false;
203
204 // Read array items
205 if (continue_reading)
206 {
207 inArray.resize(array_length);
208 for (uint32 el = 0; el < array_length && continue_reading; ++el)
209 continue_reading = OSReadData(ioStream, inArray[el]);
210 }
211
212 return continue_reading;
213}
