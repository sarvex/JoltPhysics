107 void AccumulateMeasurement(uint64 inCyclesInCallWithChildren, uint64 inCyclesInChildren)
108 {
109 mCallCounter++;
110 mTotalCyclesInCallWithChildren += inCyclesInCallWithChildren;
111 mTotalCyclesInChildren += inCyclesInChildren;
112 mMinCyclesInCallWithChildren = min(inCyclesInCallWithChildren, mMinCyclesInCallWithChildren);
113 mMaxCyclesInCallWithChildren = max(inCyclesInCallWithChildren, mMaxCyclesInCallWithChildren);
114 }
