52template <class TriangleCodec, class NodeCodec>
53class AABBTreeToBuffer
54{
55public:
57 using NodeHeader = typename NodeCodec::Header;
58
60 static const int HeaderSize = NodeCodec::HeaderSize;
61
63 static const int NumChildrenPerNode = NodeCodec::NumChildrenPerNode;
64
66 using TriangleHeader = typename TriangleCodec::TriangleHeader;
67
69 static const int TriangleHeaderSize = TriangleCodec::TriangleHeaderSize;
70
72 bool Convert(const VertexList &inVertices, const AABBTreeBuilder::Node *inRoot, AABBTreeToBufferStats &outStats, string &outError, EAABBTreeToBufferConvertMode inConvertMode = EAABBTreeToBufferConvertMode::DepthFirst)
73 {
74 const typename NodeCodec::EncodingContext node_ctx;
75 typename TriangleCodec::EncodingContext tri_ctx;
76
77 // Estimate the amount of memory required
78 uint tri_count = inRoot->GetTriangleCountInTree();
79 uint node_count = inRoot->GetNodeCount();
80 uint leaf_node_count = inRoot->GetLeafNodeCount();
81 uint nodes_size = node_ctx.GetPessimisticMemoryEstimate(node_count, leaf_node_count);
82 uint total_size = HeaderSize + TriangleHeaderSize + nodes_size + tri_ctx.GetPessimisticMemoryEstimate(tri_count);
83 mTree.reserve(total_size);
84
85 // Reset counters
86 mNodesSize = 0;
87
88 // Add headers
89 NodeHeader *header = HeaderSize > 0? mTree.Allocate<NodeHeader>() : nullptr;
90 TriangleHeader *triangle_header = TriangleHeaderSize > 0? mTree.Allocate<TriangleHeader>() : nullptr;
91
92 struct NodeData
93 {
94 const AABBTreeBuilder::Node * mNode = nullptr; // Node that this entry belongs to
95 Vec3 mNodeBoundsMin; // Quantized node bounds
96 Vec3 mNodeBoundsMax;
97 uint mNodeStart = uint(-1); // Start of node in mTree
98 uint mTriangleStart = uint(-1); // Start of the triangle data in mTree
99 uint mNumChildren = 0; // Number of children
100 uint mChildNodeStart[NumChildrenPerNode]; // Start of the children of the node in mTree
101 uint mChildTrianglesStart[NumChildrenPerNode]; // Start of the triangle data in mTree
102 uint * mParentChildNodeStart = nullptr; // Where to store mNodeStart (to patch mChildNodeStart of my parent)
103 uint * mParentTrianglesStart = nullptr; // Where to store mTriangleStart (to patch mChildTrianglesStart of my parent)
104 };
105
106 deque<NodeData *> to_process;
107 deque<NodeData *> to_process_triangles;
108 vector<NodeData> node_list;
109
110 node_list.reserve(node_count); // Needed to ensure that array is not reallocated, so we can keep pointers in the array
111
112 NodeData root;
113 root.mNode = inRoot;
114 root.mNodeBoundsMin = inRoot->mBounds.mMin;
115 root.mNodeBoundsMax = inRoot->mBounds.mMax;
116 node_list.push_back(root);
117 to_process.push_back(&node_list.back());
118
119 // Child nodes out of loop so we don't constantly realloc it
120 vector<const AABBTreeBuilder::Node *> child_nodes;
121 child_nodes.reserve(NumChildrenPerNode);
122
123 for (;;)
124 {
125 while (!to_process.empty())
126 {
127 // Get the next node to process
128 NodeData *node_data = nullptr;
129 switch (inConvertMode)
130 {
131 case EAABBTreeToBufferConvertMode::DepthFirst:
132 case EAABBTreeToBufferConvertMode::DepthFirstTrianglesLast:
133 node_data = to_process.back();
134 to_process.pop_back();
135 break;
136
137 case EAABBTreeToBufferConvertMode::BreadthFirst:
138 case EAABBTreeToBufferConvertMode::BreadthFirstTrianglesLast:
139 node_data = to_process.front();
140 to_process.pop_front();
141 break;
142
143 default:
144 JPH_ASSERT(false);
145 break;
146 }
147
148 // Due to quantization box could have become bigger, not smaller
149 JPH_ASSERT(AABox(node_data->mNodeBoundsMin, node_data->mNodeBoundsMax).Contains(node_data->mNode->mBounds), "AABBTreeToBuffer: Bounding box became smaller!");
150
151 // Collect the first NumChildrenPerNode sub-nodes in the tree
152 child_nodes.clear(); // Won't free the memory
153 node_data->mNode->GetNChildren(NumChildrenPerNode, child_nodes);
154 node_data->mNumChildren = (uint)child_nodes.size();
155
156 // Fill in default child bounds
157 Vec3 child_bounds_min[NumChildrenPerNode], child_bounds_max[NumChildrenPerNode];
158 for (size_t i = 0; i < NumChildrenPerNode; ++i)
159 if (i < child_nodes.size())
160 {
161 child_bounds_min[i] = child_nodes[i]->mBounds.mMin;
162 child_bounds_max[i] = child_nodes[i]->mBounds.mMax;
163 }
164 else
165 {
166 child_bounds_min[i] = Vec3::sZero();
167 child_bounds_max[i] = Vec3::sZero();
168 }
169
170 // Start a new node
171 uint old_size = (uint)mTree.size();
172 node_data->mNodeStart = node_ctx.NodeAllocate(node_data->mNode, node_data->mNodeBoundsMin, node_data->mNodeBoundsMax, child_nodes, child_bounds_min, child_bounds_max, mTree, outError);
173 if (node_data->mNodeStart == uint(-1))
174 return false;
175 mNodesSize += (uint)mTree.size() - old_size;
176
177 if (node_data->mNode->HasChildren())
178 {
179 for (size_t i = 0; i < child_nodes.size(); ++i)
180 {
181 // Depth first: Insert in reverse order so we process left child first when taking nodes from the back
182 size_t idx = (inConvertMode == EAABBTreeToBufferConvertMode::DepthFirst || inConvertMode == EAABBTreeToBufferConvertMode::DepthFirstTrianglesLast)?
183 child_nodes.size() - 1 - i : i;
184
185 // Due to quantization box could have become bigger, not smaller
186 JPH_ASSERT(AABox(child_bounds_min[idx], child_bounds_max[idx]).Contains(child_nodes[idx]->mBounds), "AABBTreeToBuffer: Bounding box became smaller!");
187
188 // Add child to list of nodes to be processed
189 NodeData child;
190 child.mNode = child_nodes[idx];
191 child.mNodeBoundsMin = child_bounds_min[idx];
192 child.mNodeBoundsMax = child_bounds_max[idx];
193 child.mParentChildNodeStart = &node_data->mChildNodeStart[idx];
194 child.mParentTrianglesStart = &node_data->mChildTrianglesStart[idx];
195 NodeData *old = &node_list[0];
196 node_list.push_back(child);
197 if (old != &node_list[0])
198 {
199 outError = "Internal Error: Array reallocated, memory corruption!";
200 return false;
201 }
202
203 switch (inConvertMode)
204 {
205 case EAABBTreeToBufferConvertMode::DepthFirst:
206 case EAABBTreeToBufferConvertMode::BreadthFirst:
207 to_process.push_back(&node_list.back());
208 break;
209
210 case EAABBTreeToBufferConvertMode::DepthFirstTrianglesLast:
211 case EAABBTreeToBufferConvertMode::BreadthFirstTrianglesLast:
212 if (node_list.back().mNode->HasChildren())
213 to_process.push_back(&node_list.back());
214 else
215 to_process_triangles.push_back(&node_list.back());
216 break;
217 }
218 }
219 }
220 else
221 {
222 // Add triangles
223 node_data->mTriangleStart = tri_ctx.Pack(inVertices, node_data->mNode->mTriangles, mTree, outError);
224 if (node_data->mTriangleStart == uint(-1))
225 return false;
226 }
227
228 // Patch offset into parent
229 if (node_data->mParentChildNodeStart != nullptr)
230 {
231 *node_data->mParentChildNodeStart = node_data->mNodeStart;
232 *node_data->mParentTrianglesStart = node_data->mTriangleStart;
233 }
234 }
235
236 // If we've got triangles to process, loop again with just the triangles
237 if (to_process_triangles.empty())
238 break;
239 else
240 to_process.swap(to_process_triangles);
241 }
242
243 // Finalize all nodes
244 for (NodeData &n : node_list)
245 if (!node_ctx.NodeFinalize(n.mNode, n.mNodeStart, n.mTriangleStart, n.mNumChildren, n.mChildNodeStart, n.mChildTrianglesStart, mTree, outError))
246 return false;
247
248 // Finalize the triangles
249 tri_ctx.Finalize(triangle_header, mTree);
250
251 // Get stats
252 tri_ctx.GetStats(outStats.mTriangleCodecName, outStats.mVerticesPerTriangle);
253
254 // Validate that we reserved enough memory
255 if (nodes_size < mNodesSize)
256 {
257 outError = "Internal Error: Not enough memory reserved for nodes!";
258 return false;
259 }
260 if (total_size < (uint)mTree.size())
261 {
262 outError = "Internal Error: Not enough memory reserved for triangles!";
263 return false;
264 }
265
266 // Finalize the nodes
267 if (!node_ctx.Finalize(header, inRoot, node_list[0].mNodeStart, node_list[0].mTriangleStart, outError))
268 return false;
269
270 // Shrink the tree, this will invalidate the header and triangle_header variables
271 mTree.shrink_to_fit();
272
273 // Output stats
274 outStats.mTotalSize = (uint)mTree.size();
275 outStats.mNodesSize = mNodesSize;
276 outStats.mTrianglesSize = (uint)mTree.size() - mNodesSize;
277 outStats.mBytesPerTriangle = (float)mTree.size() / tri_count;
278
279 return true;
280 }

314 uint mNodesSize;
315};
