111 inline SubShapeID MakeSubShapeIDRelativeToShape(const SubShapeID &inSubShapeID) const
112 {
113 // Take off the sub shape ID part that comes from mSubShapeIDCreator and validate that it is the same
114 SubShapeID sub_shape_id;
115 uint num_bits_written = mSubShapeIDCreator.GetNumBitsWritten();
116 JPH_IF_ENABLE_ASSERTS(uint32 root_id =) inSubShapeID.PopID(num_bits_written, sub_shape_id);
117 JPH_ASSERT(root_id == (mSubShapeIDCreator.GetID().GetValue() & ((1 << num_bits_written) - 1)));
118 return sub_shape_id;
119 }

123 inline Vec3 GetWorldSpaceSurfaceNormal(const SubShapeID &inSubShapeID, Vec3Arg inPosition) const
124 {
125 Mat44 inv_com = GetInverseCenterOfMassTransform();
126 Vec3 scale = GetShapeScale(); // See comment at ScaledShape::GetSurfaceNormal for the math behind the scaling of the normal
127 return inv_com.Multiply3x3Transposed(mShape->GetSurfaceNormal(MakeSubShapeIDRelativeToShape(inSubShapeID), (inv_com * inPosition) / scale) / scale).Normalized();
128 }
